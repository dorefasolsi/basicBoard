CCREATE TABLE BOARD(
BOARDNO INT NOT NULL AUTO_INCREMENT,
BOARDTITLE VARCHAR(255) NOT NULL,
BOARDCONTENT VARCHAR(10000) NOT NULL,
CREATEDATE DATE NOT NULL DEFAULT(CURRENT_DATE),
VIEWCOUNT INT NOT NULL DEFAULT 0,
STATUS VARCHAR(255) NOT NULL DEFAULT 'Y',
USERID VARCHAR(255),
PRIMARY KEY(BOARDNO),
FOREIGN KEY(USERID) REFERENCES USER(USERID) ON UPDATE CASCADE
)ENGINE=INNODB CHARSET=UTF8MB4;


CREATE TABLE USER(
USERID VARCHAR(255) NOT NULL,
USERPWD VARCHAR(255) NOT NULL,
USERNAME VARCHAR(255) NOT NULL,
ROLE VARCHAR(255),
LASTLOGIN DATE,
PRIMARY KEY(USERID)
)ENGINE=INNODB CHARSET=UTF8MB4;


CREATE TABLE ATTACHMENT(
FILENO INT NOT NULL AUTO_INCREMENT,
STOREDNAME VARCHAR(255) NOT NULL,
ORIGINNAME VARCHAR(255) NOT NULL,
PATH VARCHAR(255) NOT NULL,
SIZE VARCHAR(255) NOT NULL,                                      -> 삭제
UPLOADDATE DATE NOT NULL DEFAULT(CURRENT_DATE),
ATTACHED VARCHAR(255) NOT NULL,				-> Board테이블로 이동
STATUS VARCHAR(255) NOT NULL,
BOARDNO INT,
USERID VARCHAR(255),
PRIMARY KEY(FILENO),
FOREIGN KEY(BOARDNO) REFERENCES BOARD(BOARDNO) ON UPDATE CASCADE,
FOREIGN KEY(USERID) REFERENCES BOARD(USERID) ON UPDATE CASCADE
)ENGINE=INNODB CHARSET=UTF8MB4;
